Test the multiselect control, using shiftDown+mouseDown and controlDown+mouseDown
Test multiselect control
include testSetup.html  
jsxopen ${GIPATH}/shell.html?jsxapppath=${SAMPLESPATH}/test-matrix  
waitForElementPresent JsxToolbarButtonName=tbbMultiSel 5000
clickJsxElement JsxToolbarButtonName=tbbMultiSel  
waitForElementPresent JsxMatrixCellText=matrix1,Azerbaijan 5000
!-- shift key + click --
shiftKeyDown    
jsxMouseDown JsxMatrixRowId=matrix1,AX shift
assertJsxSelectedId matrix1 AX
jsxMouseDown JsxMatrixRowId=matrix1,AG shift
shiftKeyUp    
!-- control key + click --
clickJsxElement JsxToolbarButtonText=Clear*  
clickJsxElement JsxToolbarButtonText=*Select*  
waitForElementPresent JsxMatrixCellText=matrix1,Azerbaijan 5000
controlKeyDown    
mouseDown JsxMatrixRowText=matrix1,Afghanistan*
mouseDown JsxMatrixRowText=matrix1,Algeria*  
mouseDown JsxMatrixRowText=matrix1,*Angola*  
mouseDown JsxMatrixRowText=matrix1,*Austria*  
assertJsxSelectedIds matrix1 AG,AR,AO,AI
controlKeyUp    
selectJsxRecords matrix1 //record[@jsxtext="Aruba"]
