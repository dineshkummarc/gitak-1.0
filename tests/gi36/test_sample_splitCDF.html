Sample demostrate how to use XSLT to split a CDF data set.
Test JS/Mapper/XSL splitting CDF
include testSetup.html  
jsxopen ${GIPATH}/shell.html?jsxapppath=${SAMPLESPATH}/sample-splittingCDF  
waitForElementPresent JsxToolbarButtonName=tbbClear 5000
storeEval selenium.page().findByJsxName("xblkOutput1Pane") outputBlock
verifyEval storedVars['outputBlock'].getXML().selectSingleNode('//data').getChildNodes().getLength() 0
clickJsxElement JsxToolbarButtonName=tbbSplitViaJS  
verifyEval storedVars['outputBlock'].getXML().selectSingleNode('//data').getChildNodes().getLength() 2
clickJsxElement JsxToolbarButtonName=tbbClear  
verifyEval storedVars['outputBlock'].getXML().selectSingleNode('//data').getChildNodes().getLength() 0
setInterval 500
storeEval selenium.page().findByJsxName("xblkOutput2Pane") outputBlock2
clickJsxElement JsxToolbarButtonName=tbbSplitViaMapper  
verifyEval storedVars['outputBlock2'].getXML().selectSingleNode('//data').getChildNodes().getLength() 4
clickJsxElement JsxToolbarButtonName=tbbClear  
verifyEval storedVars['outputBlock2'].getXML().selectSingleNode('//data').getChildNodes().getLength() 0
clickJsxElement JsxToolbarButtonName=tbbSplitViaXSLT  
verifyEval storedVars['outputBlock'].getXML().selectSingleNode('//data').getChildNodes().getLength() 2
verifyEval storedVars['outputBlock2'].getXML().selectSingleNode('//data').getChildNodes().getLength() 4
