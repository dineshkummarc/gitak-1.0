Test Timepicker control
include testSetup.html  
open ${GIPATH}/shell.html?jsxapppath=${SAMPLESPATH}/34test-matrix  
waitForElementPresent JsxName=tpTime 5000
assertElementPresent JsxName=tpTime  
pickJsxTime tpTime 11:59:59
assertEval selenium.page().findByJsxName('tpTime').getMinutes(); 59
assertEval selenium.page().findByJsxName('tpTime').getSeconds(); 0
verifyEval selenium.page().findByJsxName('tpTime').setShowSeconds(true).repaint(); selenium.page().findByJsxName('tpTime').getShowSeconds(); 1
pickJsxTime tpTime 11:59:59
assertEval selenium.page().findByJsxName('tpTime').getSeconds(); 59
typeJsxTextbox JsxTimePickerAmPm=tpTime PM
assertEval selenium.page().findByJsxName('tpTime').getHours(); 23
typeJsxTextbox JsxTimePickerAmPm=tpTime AM
assertEval selenium.page().findByJsxName('tpTime').getHours(); 11
click JsxTimePickerHours=tpTime  
verifyEval selenium.page().findByJsxName('tpTime').setShowMillis(true).repaint(); selenium.page().findByJsxName('tpTime').getShowMillis(); 1
click JsxTimePickerSpinDown=tpTime  
assertEval selenium.page().findByJsxName('tpTime').getHours(); 10
verifyEval selenium.page().findByJsxName('tpTime').setShowSeconds(false).repaint(); selenium.page().findByJsxName('tpTime').getShowSeconds(); 0
click JsxTimePickerSpinUp=tpTime  
typeJsxTextbox JsxTimePickerAmPm=tpTime PM
verifyEval selenium.page().findByJsxName('tpTime').set24Hour(true).repaint(); selenium.page().findByJsxName('tpTime').is24Hour(); 1
pause 1500  
click JsxTimePickerSpinUp=tpTime  
assertEval selenium.page().findByJsxName('tpTime').getHours(); 0
