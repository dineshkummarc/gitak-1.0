This sample searches in a matrix list using XSL filter
Test Search matrix with XSL filter
include testSetup.html  
open ${GIPATH}/shell.html?jsxapppath=${SAMPLESPATH}/34sample-searchInList  
waitForJsxPresent list 5000
resizeJsxLayout layoutSample (--) 0,*
assertTextPresent *Search in list*  
typeJsxTextbox JsxTextboxName=searchbox Content 0
setInterval 100  
click JsxName=button //search  
//list is the matrix control
waitForJsxSelectedId list unique0
click JsxName=button //search  
waitForJsxSelectedId JsxMatrixName=list unique1
click JsxName=button //search  
waitForJsxSelectedId JsxMatrixName=list unique2
click JsxName=button //search  
waitForJsxSelectedId JsxMatrixName=list unique2
assertJsxAlertOK End of search  
setInterval 0
